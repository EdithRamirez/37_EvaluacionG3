6) Menciona las estucturas repetitivas y escribe la sintaxis de cada una:

For

for(inicialización; condición; incremento){
//sentencia
}

Ejemplo:
for(var i = 1; i <= 10; i++){
document.write(i); //muestra del 1 al 10
}

While

while(condición){
//sentencia
}

Ejemplo:
var i = 1;
while(i <= 10){
document.write(i); //muestra del 1 al 10
i++;
}

Do While

do{
//sentencia
}while(condición)

Ejemplo:
var i = 1;
do{
document.write(i); //muestra del 1 al 10
i++;
}while(i <= 10)
